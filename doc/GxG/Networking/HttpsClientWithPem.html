def initialize(pem_file_path=nil)
  unless pem_file_path.is_a?(::String)
    raise ArgumentError, "You MUST provide a path to your PEM file (as a String)."
  end
  @pem_path = pem_file_path
  @client = ::GxG::Networking::HttpClient.new()
  self
end
