def process_received(&block)
  begin
      if block.respond_to?(:call)
          count = 0
          @received_thread_safety.synchronize {
              count = @received.size
          }
          record = nil
          count.times do
              @received_thread_safety.synchronize {
                  record = @received.shift
              }
              if record
                  block.call(record)
              end
          end
          true
      else
          false
      end
  rescue Exception => the_error
      log_error({:error => the_error, :parameters => {}})
      false
  end
end
