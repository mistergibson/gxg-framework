def initialize(*args)
  params = self.process_parameters(*args)
  if params[:mode].index(:read)
    params[:mode].delete(:read)
  end
  if params[:mode].index(:readwrite)
    params[:mode].delete(:readwrite)
  end
  unless params[:mode].index(:write)
    params[:mode] << :write
  end
  super(params)
  self
end
